#1 Задание

SELECT c.login,
       COUNT(o."inDelivery") AS Orders_count
FROM "Couriers" AS c
LEFT OUTER JOIN "Orders" AS O ON c.id=o."courierId"
WHERE o."inDelivery" = true
GROUP BY c.login;
